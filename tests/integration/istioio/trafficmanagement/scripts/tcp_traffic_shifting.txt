set -u pipefail #return the status of last execution or 0 in case of no status
set -e #forces an early exit when a command executes and return non-zero status

#step 1

# $snippet deploy_with_automatic_sidecar_injection syntax="bash" outputis="text"
$ kubectl label namespace default istio-injection=enabled
$ kubectl apply -f @samples/tcp-echo/tcp-echo-services.yaml@
# $verify
service/tcp-echo created
deployment.apps/tcp-echo-v1 created
deployment.apps/tcp-echo-v2 created
# $endsnippet

# Step 2: route all TCP traffic to v1 version of tcp-echos service

# $snippet route_all_traffic_to_v1 syntax="bash" outputis="text"
$ kubectl apply -f samples/tcp-echo/tcp-echo-all-v1.yaml@
# $endsnippet

# Step 3: Confirm that the tcp-echo service is up and running:

# $snippet define_ingress_port_variable syntax="bash"
$ export INGRESS_PORT=$(kubectl -n istio-system get service istio-ingressgateway -o jsonpath='{.spec.ports[?(@.name=="tcp")].port}')
# $endsnippet
